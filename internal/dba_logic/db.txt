CREATE TABLE IF NOT EXISTS "Task" (
    id SERIAL PRIMARY KEY,
    title TEXT NOT NULL,
    description TEXT,
    status BOOLEAN NOT NULL,  -- Булевый статус выполнения задачи
    added_time TIMESTAMP NOT NULL,  -- Дата и время без учета часового пояса
    images Bytea,  -- Массив строк Base64 для хранения изображений
    user_id INTEGER NOT NULL,  -- Внешний ключ для связи с таблицей User
    FOREIGN KEY (user_id) REFERENCES "User" (id)
);
